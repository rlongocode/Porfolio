116 private void operation_check(String operation) {
117     //General Case For Empty Sets
118     if (operation.equals("min") || operation.equals("max") || operation.equals("sortasc")|| operation.equals("sortdes")){
119       if (this.sete.isEmpty()) {
120         JOptionPane.showMessageDialog(this, "The set is empty.", "Empty Set Error", JOptionPane.INFORMATION_MESSAGE);
121         
122         return;
123       }
124     }
125     
126     //Inserts Number to list, made a sencond list for Asceding and Decending sort with checkboxes
127     try {
128      
129       if (operation.equals("insert")) {
130         String text = input.getText();
131         int the_num = Integer.parseInt(text);
132         this.sete.insert(the_num);
133         
134         this.sort_list.insert(the_num);
135         this.sort_listb.insert(the_num);
136         this.display.setText(this.sete.toString());
137         this.input.setText("");
138       } 
139 
140       else if (operation.equals("sortasc")){
141           this.sort_list.upsorter();
142           this.results.setText(this.sort_list.toString());
143       }
144       
145       else if (operation.equals("sortdes")){
146           this.sort_list.downsorter();
147           this.results.setText(this.sort_list.toString());
148       }
149       
150       else if (operation.equals("sortascdes")){
151           this.sort_list.upsorter();
152           this.sort_listb.equals(this.sort_list);
153           this.sort_listb.upsorter();
154           this.sort_list.downsorter();
155           this.results.setText(this.sort_listb.toString() + this.sort_list.toString());
156       }
157       
158       else if (operation.equals("max")) {
159         this.maxi = this.sete.maxvoid();
160         this.results.setText("Max number: " + Integer.toString(this.maxi));
161         }
162       
163       else if (operation.equals("min")) {
164         this.mini = this.sete.minvoid();
165         this.results.setText("Min number: " + Integer.toString(this.mini));
166         }
167     } 
168     //Exception for No Number or No Input
169     catch (NumberFormatException exception) {
170       JOptionPane.showMessageDialog(this, "Enter a number", "Error: Not A Number", JOptionPane.INFORMATION_MESSAGE);
171     } 
172     catch (Exception exception) {
173       JOptionPane.showMessageDialog(this, "Empty, please add numbers.", "Error: Empty Set", JOptionPane.INFORMATION_MESSAGE);
174     } 
175 }

177 //Checks it works properly
178 public void actionPerformed(ActionEvent event) {
179     if (event.getSource() == this.inserte) {
180       operation_check("insert");
181     } 
182     
183     else if (event.getSource() == this.maxnum) {
184       operation_check("max");
185     } 
186     
187     else if (event.getSource() == this.sortasc) {
188       operation_check("sortasc");
189     } 
190     
191     else if (event.getSource() == this.sortdes) {
192       operation_check("sortdes");
193     } 
194     
195     else if (event.getSource() == this.minnum) {
196       operation_check("min");
197     } 
198     
199     else if (event.getSource() == this.sortpress) {
200        if (this.sortasc.isSelected()) {
201                 operation_check("sortasc");
202             }
203        if (this.sortdes.isSelected()) {
204                 operation_check("sortdes");
205             }
206        if (this.sortasc.isSelected() && this.sortdes.isSelected()) {
207                 operation_check("sortascdes");
208             }
209     } 
210     
211     else if (event.getSource() == this.cleare) {
212       
213       this.display.setText("");
214       this.results.setText("");
215       this.sort_list.equals("");
216       this.sort_listb.equals("");
217 
218       try {
219         this.sete.clear();
220         this.sort_list.clear();
221         this.sort_listb.clear();
222       } 
223 
224       catch (Exception exception) {}
225     } 
226   }

229 private class IntSet {
230     public Vector<Integer> vec = new Vector<>();
231     public int maxi;
232     public int mini;
233     
234     private IntSet() {
235     }
236     
237     public void insert(int no) {
238       vec.addElement(no);
239     }
240     
241     public int maxvoid(){
242         return Collections.max(vec);
243     }
244     
245     public int minvoid(){
246         return Collections.min(vec);
247     }
248 
249     public void upsorter()
250     {
251         Collections.sort(vec);
252     }
253     
254     public void downsorter()
255     {
256         Collections.sort(vec);
257         Collections.reverse(vec);
258     }
259     
260     public int getSize() {
261       return vec.size();
262     }
263     
264     public boolean isEmpty() {
265       return (vec.isEmpty());
266     }
267     
268     public void clear() throws Exception {
269       vec.removeAllElements();
270     }
271     
272     public String toString() {
273       return vec.toString();
274     }

276   }
